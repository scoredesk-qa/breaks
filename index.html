-- Table for break logs create table if not exists bt_logs ( id uuid default gen_random_uuid() primary key, agent_id uuid references auth.users(id) on delete cascade, agent_name text not null, start_time bigint not null, end_time bigint, break_type text, created_at timestamptz default now() ); -- Table for admin settings create table if not exists bt_settings ( key text primary key, value integer not null default 0 ); -- Enable RLS alter table bt_logs enable row level security; alter table bt_settings enable row level security; -- Policies: agents manage their own logs create policy "agents_view_own" on bt_logs for select using (auth.uid() = agent_id); create policy "agents_insert_own" on bt_logs for insert with check (auth.uid() = agent_id); create policy "agents_update_own" on bt_logs for update using (auth.uid() = agent_id); -- Admins can view/delete all logs create policy "admins_view_all" on bt_logs for select using (auth.jwt() ->> 'role' = 'admin'); create policy "admins_delete_all" on bt_logs for delete using (auth.jwt() ->> 'role' = 'admin'); -- Only admins can modify settings create policy "admins_manage_settings" on bt_settings for all using (auth.jwt() ->> 'role' = 'admin');
